1   package model;
2   
3   import java.util.Observable;
4   import java.util.Observer;
5   
6   /**
7    * This class describe how to use a map
8    * 
9    * @author Vincent Linck
10   * @version 1.1
11   */
12  public class LorannMap extends Observable implements ILorannMap {
13  	
14  	/**
15  	 * The map that the game use
16  	 */
17  	private IEntity[][] map;
18  	
19  	/**
20  	 * the width of the map
21  	 */
22  	private int width;
23  	
24  	/**
25  	 * the height of the map
26  	 */
27  	private int height;
28  	
29  	/**
30  	 * Instantiate the map
31  	 * @param width
32  	 * @param height
33  	 */
34  	public LorannMap(int width, int height) {
35  		this.width = width;
36  		this.height = height;
37  		map = new IEntity[this.height][this.width];
38  		
39  		for (int y = 0; y < this.height; y++) {
40  			for (int x = 0; x < this.width; x++) {
41  				map[y][x] = null;
42  			}
43  		}
44  	}
45  	
46  	/**
47  	 * Get an entity on the map
48  	 * @param x
49  	 * @param y
50  	 * @return
51  	 */
52  	public IEntity getOnMap(int x, int y) {
53  		if (x >= 0 && x < this.width && y >= 0 && y < this.height) {
54  			return map[y][x];
55  		}
56  		return new Entity("../sprite/ground.png", Permeability.BLOCKING);
57  	}
58  	
59  	/**
60  	 * Set an entity on the map
61  	 * @param entity
62  	 * @param x
63  	 * @param y
64  	 */
65  	public void setOnMap(IEntity entity, int x, int y) {
66  		if (x >= 0 && x < this.width && y >= 0 && y < this.height) {
67  			map[y][x] = entity;
68  		}
69  	}
70  	
71  	/**
72       * Sets the mobile has changed.
73       */
74  	public final void setMobileHasChanged() {
75          this.setChanged();
76          this.notifyObservers();
77      }
78      
79      @Override
80      public void addObserver(Observer observer) {
81      	super.addObserver(observer);
82      }
83  
84  }
